local Key = game.CreatorId

local Users = {1415011184, 2210775622}
local License

local function Load(Player)
	for u, users in pairs(Users) do
		if users == Player then
			License = users
		end
	end
	if Key == License then
		return true
	end
end

local function Denied()
	for sc, scripts in ipairs(game.ServerScriptService:GetDescendants()) do
		if scripts:IsA("Script") or scripts:IsA("LocalScript") then
			scripts:Destroy()
		end
	end
	
	for p, players in ipairs(game.Players:GetChildren()) do
		for sg, guis in ipairs(game.Players[players].PlayerGui:GetDescendants()) do
			if guis:IsA("ScreenGui") or guis:IsA("LocalScript") then
				guis:Destroy()
			end
		end
	end
	
	for c, child in ipairs(game.Players:GetChildren()) do
		child:Kick("Buy one license! Don't try crack it.")
	end
end

game:GetService("Players").PlayerAdded:Connect(function(Player)
	print("Checking...")
	print("Using License Key: "..Key)
	if Load(game.CreatorId) then
		print("Access Granted!")
	else
		print("Access Denied!")
		Denied()
	end
end)
